Pivot wider so cases and death are own category and calculate

gender_wide <- gender_latest |>
  pivot_wider(
    names_from = outcomes,
    values_from = count
  )
gender_rates <- gender_wide |>
  mutate(
    death_rate = deaths / cases
  )
gender_rates <- gender_rates |>
  mutate(
    death_rate_percent = death_rate * 100
  )

create a visualization

ggplot(gender_rates, aes(x = group, y = death_rate_percent)) +
  geom_col() +
  labs(
    title = "COVID-19 Death Rate by Gender",
    x = "Gender",
